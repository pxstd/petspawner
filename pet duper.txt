--ðŸŒˆ RGB Color Generator
local function getRainbowColor(hueShift)
    return Color3.fromHSV((tick() + hueShift) % 5 / 5, 1, 1)
end

-- UI Setup
local gui = Instance.new("ScreenGui", playerGui)
gui.Name = "TONXHUB_UI"
gui.ResetOnSpawn = false
gui.IgnoreGuiInset = true -- Important for mobile

-- Main Frame (draggable)
local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 280, 0, 150)
frame.Position = UDim2.new(0.3, 0, 0.3, 0) -- Start position
frame.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
frame.BorderSizePixel = 0
frame.AnchorPoint = Vector2.new(0, 0)
frame.Parent = gui

-- Make frame draggable
local dragStartPos
local frameStartPos

local function startDrag(input)
    dragStartPos = input.Position
    frameStartPos = frame.Position
    input.Changed:Connect(function()
        if input.UserInputState == Enum.UserInputState.End then
            dragStartPos = nil
        end
    end)
end

local function updateDrag(input)
    if dragStartPos then
        local delta = input.Position - dragStartPos
        frame.Position = UDim2.new(
            frameStartPos.X.Scale, 
            frameStartPos.X.Offset + delta.X,
            frameStartPos.Y.Scale,
            frameStartPos.Y.Offset + delta.Y
        )
    end
end

frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
        startDrag(input)
    end
end)

frame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement then
        updateDrag(input)
    end
end)

local function animateClick(button)

    task.wait(0.1)
    button.Size = UDim2.new(0.8, 0, 0, 35)
end

-- Dupe Function
local function dupePet()
    local pet = nil
    for _, v in ipairs(character:GetChildren()) do
        if v:IsA("Tool") then -- Assuming pets are Tools in the game
            pet = v
            break
        end
    end

    if not pet then 
        heldPetLabel.Text = "Held Pet: None (Hold a pet!)"
        task.wait(1)
        heldPetLabel.Text = "Held Pet: None"
        return 
    end

    local clone = pet:Clone()
    clone.Parent = backpack

    -- Visual feedback
    dupeBtn.Text = "DUPE SUCCESS!"
    task.wait(0.5)
    dupeBtn.Text = "DUPE PET"
end

-- Button Clicked
dupeBtn.MouseButton1Click:Connect(function()
    animateClick(dupeBtn)
    dupePet()
end)

-- Touch support for mobile
dupeBtn.TouchTap:Connect(function()
    animateClick(dupeBtn)
    dupePet()
end)
